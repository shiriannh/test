PDepend\Source\AST\ASTCastExpression

This class represents a cast-expression node.

<code>
// ----------
$foo = (int) $bar;
// ----------

// -----------
$foo = (bool) $bar;
// -----------

// ------------
$foo = (array) $bar;
// ------------

// ------------
$foo = (unset) $bar;
// ------------

// -------------
$foo = (double) $bar;
// -------------

// -------------
$foo = (string) $bar;
// -------------

// -------------
$foo = (object) $bar;
// -------------
</code>
