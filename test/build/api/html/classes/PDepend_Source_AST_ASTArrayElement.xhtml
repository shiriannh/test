PDepend\Source\AST\ASTArrayElement

This class represents a single array element expression.

<code>
// _ ______ __________________________
// __ ___ __________
$array = array( 1, 2 => 3, array( $a, &$b, 1 => &$c ) );

// _ ______ _____________________
// __ ___ ________
$array = [ 1, 2 => 3, [ $a, &$b, 1 => &$c ] ];
</code>
